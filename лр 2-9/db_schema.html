📄 DDL — SQL создания таблиц

-- База данных
CREATE DATABASE IF NOT EXISTS `aquatica`
  CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

USE `aquatica`;

-- 1. Услуги
CREATE TABLE IF NOT EXISTS services (
  id         INT AUTO_INCREMENT PRIMARY KEY,
  title      VARCHAR(255)               NOT NULL,
  category   ENUM('courses','tours')    NOT NULL DEFAULT 'courses',
  price      INT                        NOT NULL DEFAULT 0,
  descr      TEXT,
  active     TINYINT(1)                 NOT NULL DEFAULT 1,
  created_at DATETIME                   NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 2. Заказы
CREATE TABLE IF NOT EXISTS orders (
  id           INT AUTO_INCREMENT PRIMARY KEY,
  client_name  VARCHAR(150)  NOT NULL,
  client_email VARCHAR(150)  NOT NULL,
  service      VARCHAR(255)  NOT NULL,
  food         VARCHAR(100),
  options      VARCHAR(255),
  doc_name     VARCHAR(255),
  mail_sent    TINYINT(1)    NOT NULL DEFAULT 0,
  created_at   DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 3. Отзывы
CREATE TABLE IF NOT EXISTS reviews (
  id         INT AUTO_INCREMENT PRIMARY KEY,
  author     VARCHAR(100)  NOT NULL,
  body       TEXT          NOT NULL,
  created_at DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 4. Пользователи
CREATE TABLE IF NOT EXISTS users (
  id         INT AUTO_INCREMENT PRIMARY KEY,
  email      VARCHAR(150)           NOT NULL UNIQUE,
  pass_hash  VARCHAR(255)           NOT NULL,
  name       VARCHAR(100)           NOT NULL,
  role       ENUM('user','admin')   NOT NULL DEFAULT 'user',
  created_at DATETIME               NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

🚀 Порядок запуска

1. Загрузите все три файла на сервер:
   📄 install.php   — установщик БД (запустить 1 раз)
   📄 index.php     — основной сайт
   📄 db_schema.html — эта схема (можно удалить)

2. Откройте в браузере:
   http://ваш-сайт/install.php

3. Установщик создаст:
   ✅ База данных  aquatica
   ✅ Пользователь aquatica_user / Aquatica_2025!
   ✅ Таблицы: services, orders, reviews, users
   ✅ Демо-данные (7 услуг)
   ✅ Файл config.php с реквизитами

4. Настройте Gmail SMTP в index.php:
   Gmail → Настройки → Безопасность → Пароли приложений
   Создайте пароль для "Почта / Windows/Mac"
   Вставьте 16 символов в SMTP_PASS
